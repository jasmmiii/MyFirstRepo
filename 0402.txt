function fetchUnAssignedReissueCardData(reissueNo, currentBillingDay) {
    $('#createUnit').text('');
    $('#reissueNo').text('');
    $('#customerId').text('');
    $('#customerName').text('');
    $('#cardNo').text('');
    $('#billingDay').text('');
    $('#reissueCardNote').text('');

    callApi({
        url: contextPath + "api/card/unassigned/" + reissueNo,
        method: "GET",
        success: function(response) {
//            console.log("單筆資料回傳:", response);

            if (response.success && response.data) {
                const modalData = response.data;

                $('#createUnit').text(modalData.createUnit);
                $('#reissueNo').text(modalData.reissueNo);
                $('#customerId').text(maskLastFourDigits(modalData.customerId));
                $('#customerName').text(maskName(modalData.customerName));
                $('#cardNo').text(maskCardNo(modalData.cardNo));
                $('#billingDay').text(modalData.billingDay);
                $('#reissueCardNote').text(modalData.reissueCardNote);

//                currentBillingDay = modalData.billingDay;
            }
        },
        error: function(xhr) {
            showErrorMessage("獲取單筆資料失敗:", xhr);
        }
    });

//    return currentBillingDay;
}

// 結帳日檢核
//let currentBillingDay = null;
function checkBillingDay(dateObj, billingDay){
    const dateOfToday = dateObj.getDate();
    const dayOfWeek = dateObj.getDay();

    if(dayOfWeek === 5){
        const tmrDate = new Date(dateObj.getFullYear(), dateObj.getMonth(), dateOfToday + 1).getDate();
        const dayAfterDate = new Date(dateObj.getFullYear(), dateObj.getMonth(), dateOfToday + 2).getDate();

        if (tmrDate === billingDay || dayAfterDate === billingDay){ return false; }
    }
    return true;
}
